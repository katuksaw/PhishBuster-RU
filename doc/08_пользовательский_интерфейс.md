# PhishBuster RU — Пользовательский интерфейс popup

## Дизайн-концепция

### Принципы
1. **Минимализм** — пользователь видит результат за 1 секунду
2. **Цветовое кодирование** — зелёный/жёлтый/красный — интуитивно понятно
3. **Информативность** — показываем не только вердикт, но и причины
4. **Профессионализм** — дизайн выглядит как настоящий продукт, не школьный прототип

### Цветовая палитра


## Реализация popup.js

```javascript
// popup.js — логика интерфейса popup

document.addEventListener('DOMContentLoaded', () => {
  loadResult();
});

async function loadResult() {
  try {
    // Получаем активную вкладку
    const [tab] = await chrome.tabs.query({ active: true, currentWindow: true });
    
    if (!tab) {
      showError();
      return;
    }
    
    // Запрашиваем результат у background.js
    chrome.runtime.sendMessage(
      { action: "getResult", tabId: tab.id },
      (response) => {
        if (chrome.runtime.lastError) {
          console.error('Ошибка:', chrome.runtime.lastError);
          showError();
          return;
        }
        
        if (response && response.score !== undefined) {
          showResult(response, tab.url);
        } else {
          showNeutral(tab.url);
        }
      }
    );
  } catch (error) {
    console.error('Ошибка в popup:', error);
    showError();
  }
}

function showResult(result, url) {
  // Скрываем loading
  document.getElementById('loadingState').style.display = 'none';
  document.getElementById('resultState').style.display = 'block';
  
  // Score ring
  const scoreRing = document.getElementById('scoreRing');
  scoreRing.className = `score-ring ${result.verdict}`;
  document.getElementById('scoreValue').textContent = result.score;
  
  // Verdict text
  const verdictText = document.getElementById('verdictText');
  const verdictLabels = {
    safe: "✅ Безопасно",
    warning: "⚠️ Подозрительно",
    danger: "🚨 ФИШИНГ!",
    neutral: "ℹ️ Неизвестный сайт"
  };
  verdictText.textContent = verdictLabels[result.verdict] || "—";
  verdictText.className = `verdict-text ${result.verdict}`;
  
  // URL
  document.getElementById('urlText').textContent = url;
  
  // Reasons
  const reasonsSection = document.getElementById('reasonsSection');
  const reasonsList = document.getElementById('reasonsList');
  reasonsList.innerHTML = '';
  
  if (result.reasons && result.reasons.length > 0) {
    reasonsSection.style.display = 'block';
    result.reasons.forEach(reason => {
      const item = document.createElement('div');
      item.className = `reason-item ${result.verdict}`;
      
      const icons = {
        safe: '✅',
        warning: '⚠️',
        danger: '🔴',
        neutral: 'ℹ️'
      };
      
      item.innerHTML = `
        <span class="reason-icon">${icons[result.verdict] || '•'}</span>
        <span>${reason}</span>
      `;
      reasonsList.appendChild(item);
    });
  } else {
    reasonsSection.style.display = 'none';
  }
  
  // Service
  if (result.serviceName) {
    document.getElementById('serviceSection').style.display = 'block';
    document.getElementById('serviceName').textContent = result.serviceName;
  }
  
  // Footer
  document.getElementById('rulesVersion').textContent = `База: v${result.rulesVersion || '—'}`;
  document.getElementById('cacheInfo').textContent = result.cached ? 'из кэша' : 'свежий анализ';
}

function showNeutral(url) {
  showResult({
    score: 0,
    verdict: "neutral",
    reasons: ["Сайт не похож на известные сервисы"],
    serviceName: null,
    rulesVersion: "—"
  }, url);
}

function showError() {
  document.getElementById('loadingState').style.display = 'none';
  document.getElementById('errorState').style.display = 'block';
}
```

## Обоснование дизайнерских решений

| Элемент | Решение | Обоснование |
|---|---|---|
| Ширина 360px | Стандартная ширина popup Chrome | Оптимально для читаемости |
| Кольцо score | Визуальный якорь внимания | Пользователь мгновенно видит уровень угрозы |
| Градиент на кольце | От тёмного к светлому | Придаёт глубину, выглядит современно |
| Тень на кольце | `box-shadow` | Выделяет главный элемент интерфейса |
| Иконки-эмодзи | Вместо иконок | Не нужны внешние ресурсы, работают в sandbox |
| Цветовые метки | Зелёный/жёлтый/красный | Универсальная семантика (светофор) |
| Footer с версией | Внизу | Информация для тех, кому интересно |
| Loading spinner | Анимация при загрузке | Пользователь понимает, что идёт анализ |
| `word-break: break-all` для URL | Длинные URL не ломают вёрстку | Практическая необходимость |
